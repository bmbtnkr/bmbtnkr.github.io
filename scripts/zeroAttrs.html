import maya.cmds as cmds
import maya.mel as mel


def reset_attrs():
    sel = cmds.ls(sl=True)
    
    if len(sel) == 0:
        raise StandardError, 'Error: Please select an object'    
        
    obj_list = sel[:]
    xyz = ['X','Y','Z']
    non_zero_attrs = ['sx', 'sy', 'sz', 'v']
    channelBox = mel.eval('global string $gChannelBoxName; $temp=$gChannelBoxName;')
    attrs = cmds.channelBox(channelBox, q=True, sma=True)
    
    if attrs == None:
        for axis in xyz:
            for obj in obj_list:
                cmds.setAttr((obj +'.translate'+axis), 0)
                cmds.setAttr((obj +'.rotate'+axis), 0)  
    else:
        for attr in attrs:
            for obj in obj_list:
                if attr in non_zero_attrs:
                    continue
                else:
                    cmds.setAttr(obj+'.'+attr, 0)
            
reset_attrs()

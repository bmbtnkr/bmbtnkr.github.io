import maya.cmds as cmds

def orient_snap():
    sel = cmds.ls(sl = True)
    
    if len(sel) < 2:
        raise StandardError, 'Error: Please select at least 2 objects' 
        
    obj_list = sel[:-1]
    target = sel[-1]
    ori_constraint_list = []
    
    for objs in obj_list:
      ori_constraint = cmds.orientConstraint(target, objs, mo=0)
      ori_constraint_list.append(ori_constraint)
  
    for temp_constraint in ori_constraint_list:  
      cmds.delete(temp_constraint)
      
orient_snap()

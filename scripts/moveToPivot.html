# moveToPivot
# Will match a selected object to the pivot of another object

# Import Mel package
from maya.mel import *
import maya.cmds as cmds

def moveToPivot():
    #Get selected objects
    currentSel = cmds.ls(sl = True)
    base = currentSel[1]
    target = currentSel[0]
    
    worldSpaceBase = cmds.xform(base, q=True, ws=True, rp=True)
    cmds.setAttr((target+".translateX"), worldSpaceBase[0])
    cmds.setAttr((target+".translateY"), worldSpaceBase[1])
    cmds.setAttr((target+".translateZ"), worldSpaceBase[2])

#Run moveToPivot    
moveToPivot()
